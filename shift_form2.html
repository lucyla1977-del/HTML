function doPost(e) { var ss = SpreadsheetApp.openById("11MWKv3GAAJTtZUPvH0LVyjTU4PSAvggQ8Le04DLRRc8"); var data = JSON.parse(e.postData.contents); var sheetName = data.Week_Start; var startParts = sheetName.split("-"); var startDate = new Date(+startParts[0], +startParts[1] - 1, +startParts[2]); var dayNames = ["Sunday","Monday","Tuesday","Wednesday","Thursday","Friday","Saturday"]; var dayHeaders = []; for (var i = 0; i < 7; i++) { var d = new Date(startDate.getFullYear(), startDate.getMonth(), startDate.getDate() + i); dayHeaders.push(dayNames[i] + " " + String(d.getDate()).padStart(2,"0") + "/" + String(d.getMonth()+1).padStart(2,"0") + "/" + d.getFullYear()); } var headers = ["Employee_Name","Max_Shifts"].concat(dayHeaders).concat(["Notes"]); var sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); sheet.getRange(1,1,1,headers.length).setValues([headers]); sheet.getRange(1,1,1,headers.length).setFontWeight("bold"); sheet.setFrozenRows(1); } var lastRow = sheet.getLastRow(); var existingRow = -1; if (lastRow > 1) { var names = sheet.getRange(2,1,lastRow-1,1).getValues(); for (var i = 0; i < names.length; i++) { if (names[i][0] === data.Employee_Name) { existingRow = i+2; break; } } } var row = [data.Employee_Name, data.Max_Shifts]; for (var i = 0; i < 7; i++) row.push(data[dayNames[i]] || ""); row.push(data.Notes || ""); if (existingRow > 0) sheet.getRange(existingRow,1,1,row.length).setValues([row]); else sheet.appendRow(row); return ContentService.createTextOutput(JSON.stringify({status:"ok"})).setMimeType(ContentService.MimeType.JSON); }
